document.getElementById("yesBtn").addEventListener("click", function() {
    document.body.style.backgroundColor = "#ffaad5";
    alert("Thank you! You're my Valentine ❤️");
    showHeart();
});

document.getElementById("noBtn").addEventListener("mouseover", function() {
    let x = Math.random() * window.innerWidth;
    let y = Math.random() * window.innerHeight;

    document.getElementById("noBtn").style.left = x + 'px';
    document.getElementById("noBtn").style.top = y + 'px';
});

function showHeart() {
    const heart = document.getElementById("heart");
    heart.classList.remove("hidden");
    setTimeout(function() {
        heart.classList.add("hidden");
    }, 2500); // Hide heart after 2.5 seconds
}
